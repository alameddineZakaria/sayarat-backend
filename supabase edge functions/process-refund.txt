import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

export const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type'
};

Deno.serve(async (req) => {
  // Handle CORS preflight
  if (req.method === 'OPTIONS') {
    return new Response('ok', { headers: corsHeaders });
  }

  try {
    const supabaseUrl = Deno.env.get('SUPABASE_URL')!;
    const supabaseServiceKey = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!;
    const supabase = createClient(supabaseUrl, supabaseServiceKey);

    // Get authorization header
    const authHeader = req.headers.get('Authorization');
    if (!authHeader) {
      return new Response(JSON.stringify({ error: 'No authorization header' }), {
        status: 401,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    // Verify the user is an admin
    const { data: { user }, error: authError } = await supabase.auth.getUser(
      authHeader.replace('Bearer ', '')
    );

    if (authError || !user) {
      return new Response(JSON.stringify({ error: 'Unauthorized' }), {
        status: 401,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    // Check admin status
    const { data: adminData, error: adminError } = await supabase
      .from('admin_users')
      .select('role, permissions')
      .eq('user_id', user.id)
      .eq('is_active', true)
      .single();

    if (adminError || !adminData) {
      return new Response(JSON.stringify({ error: 'Admin access required' }), {
        status: 403,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    const permissions = adminData.permissions as { manage_payments?: boolean } || {};
    if (adminData.role !== 'super_admin' && !permissions.manage_payments) {
      return new Response(JSON.stringify({ error: 'Payment management permission required' }), {
        status: 403,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    // Parse request body
    const { 
      payment_id, 
      stripe_payment_id, 
      reason, 
      amount,
      refund_type = 'full' // 'full' or 'partial'
    } = await req.json();

    if (!payment_id) {
      return new Response(JSON.stringify({ error: 'payment_id is required' }), {
        status: 400,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    // Get the purchase record
    const { data: purchase, error: purchaseError } = await supabase
      .from('purchases')
      .select(`
        *,
        user:profiles!user_id(id, email, full_name)
      `)
      .eq('id', payment_id)
      .single();

    if (purchaseError || !purchase) {
      return new Response(JSON.stringify({ error: 'Purchase not found' }), {
        status: 404,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    // Check if already refunded
    if (purchase.refund_status === 'refunded') {
      return new Response(JSON.stringify({ error: 'This purchase has already been refunded' }), {
        status: 400,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    // Calculate refund amount
    const refundAmount = refund_type === 'partial' && amount 
      ? Math.min(amount, purchase.amount) 
      : purchase.amount;

    // Get Stripe API key from environment
    const stripeApiKey = Deno.env.get('GATEWAY_API_KEY');
    
    let stripeRefund = null;
    let refundId = `refund_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;

    // If we have a Stripe payment ID and API key, process the refund via Stripe
    if (stripeApiKey && (stripe_payment_id || purchase.transaction_id?.startsWith('cs_'))) {
      const paymentIntentId = stripe_payment_id || purchase.metadata?.payment_intent;
      
      if (paymentIntentId && paymentIntentId.startsWith('pi_')) {
        try {
          // Create refund via Stripe API
          const refundResponse = await fetch('https://api.stripe.com/v1/refunds', {
            method: 'POST',
            headers: {
              'Authorization': `Bearer ${stripeApiKey}`,
              'Content-Type': 'application/x-www-form-urlencoded',
            },
            body: new URLSearchParams({
              payment_intent: paymentIntentId,
              amount: Math.round(refundAmount * 100).toString(), // Convert to cents
              reason: 'requested_by_customer',
              ...(reason && { metadata: JSON.stringify({ admin_reason: reason }) })
            }).toString()
          });

          if (refundResponse.ok) {
            stripeRefund = await refundResponse.json();
            refundId = stripeRefund.id;
            console.log('Stripe refund created:', refundId);
          } else {
            const errorData = await refundResponse.json();
            console.error('Stripe refund error:', errorData);
            // Continue with local refund tracking even if Stripe fails
          }
        } catch (stripeError) {
          console.error('Error calling Stripe API:', stripeError);
          // Continue with local refund tracking
        }
      }
    }

    // Update the purchase record with refund info
    const { error: updateError } = await supabase
      .from('purchases')
      .update({
        refund_status: refund_type === 'partial' ? 'partially_refunded' : 'refunded',
        refund_amount: refundAmount,
        refund_reason: reason,
        refunded_at: new Date().toISOString(),
        refund_id: refundId,
        status: 'refunded',
        metadata: {
          ...purchase.metadata,
          refund: {
            type: refund_type,
            amount: refundAmount,
            reason: reason,
            processed_by: user.id,
            processed_at: new Date().toISOString(),
            stripe_refund_id: stripeRefund?.id || null
          }
        }
      })
      .eq('id', payment_id);

    if (updateError) {
      console.error('Error updating purchase:', updateError);
      return new Response(JSON.stringify({ error: 'Failed to update purchase record' }), {
        status: 500,
        headers: { 'Content-Type': 'application/json', ...corsHeaders }
      });
    }

    // If this was a subscription purchase, handle subscription cancellation
    if (purchase.product_id?.includes('subscription') && refund_type === 'full') {
      const { error: subError } = await supabase
        .from('subscriptions')
        .update({
          status: 'cancelled',
          updated_at: new Date().toISOString()
        })
        .eq('user_id', purchase.user_id);

      if (!subError) {
        // Update user profile
        await supabase
          .from('profiles')
          .update({
            subscription_status: 'cancelled',
            is_dealer: false
          })
          .eq('id', purchase.user_id);
      }
    }

    // If this was a boost purchase, deactivate the boost
    if (purchase.product_id?.includes('boost') && purchase.metadata?.listing_id) {
      await supabase
        .from('vehicles')
        .update({
          is_boosted: false,
          boost_expires_at: null,
          boost_level: null
        })
        .eq('id', purchase.metadata.listing_id);
    }

    // Send refund notification email
    const userEmail = purchase.user?.email || purchase.metadata?.customer_email;
    if (userEmail) {
      try {
        await fetch(`${supabaseUrl}/functions/v1/send-email-receipt`, {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${supabaseServiceKey}`,
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({
            email: userEmail,
            customerName: purchase.user?.full_name,
            productId: purchase.product_id,
            transactionId: refundId,
            amount: refundAmount,
            currency: purchase.currency || 'USD',
            purchaseDate: new Date().toISOString(),
            isRefund: true,
            refundReason: reason,
            originalTransactionId: purchase.transaction_id
          }),
        });
      } catch (emailError) {
        console.error('Error sending refund email:', emailError);
        // Don't fail the refund if email fails
      }
    }

    return new Response(JSON.stringify({
      success: true,
      refund_id: refundId,
      refund_amount: refundAmount,
      refund_type: refund_type,
      stripe_refund: stripeRefund ? {
        id: stripeRefund.id,
        status: stripeRefund.status
      } : null
    }), {
      status: 200,
      headers: { 'Content-Type': 'application/json', ...corsHeaders }
    });

  } catch (error: any) {
    console.error('Refund processing error:', error);
    return new Response(JSON.stringify({ error: error.message }), {
      status: 500,
      headers: { 'Content-Type': 'application/json', ...corsHeaders }
    });
  }
});
